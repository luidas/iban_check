class TestIbanCheck(unittest.TestCase):
    def test_valid_ibans(self):
        """
        Test that it can find a valid iban
        """
        for iban in valid:
            with self.subTest():
                self.assertTrue(algorithm.check_iban(iban))

    def test_invalid_ibans(self):
        """
        Test that it can find an invalid iban
        """
        for iban in invalid:
            with self.subTest():
                self.assertFalse(algorithm.check_iban(iban))

    def test_wrong_type(self):
        """
        Test that it doesn't accept wrong types
        """
        for iban in wrong_type:
            with self.subTest():
                self.assertFalse(algorithm.check_iban(iban))
